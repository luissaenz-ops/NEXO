        async function fetchGeminiResponse(promptText) {

            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

            

            // System instruction defining the personality and goal of the LLM

            const systemPrompt = `Eres el "Asistente de Carreras NEXO", una Inteligencia Artificial creada por el partido estudiantil NEXO de Keiser University (liderado por Luis Larios). 

            Tu misión es ayudar a los estudiantes a reducir la brecha entre su formación académica y el mundo laboral. 

            Tono: Amigable, motivador, usas 'vos' (estilo nicaragüense/latinoamericano), profesional pero juvenil.

            Instrucción: El usuario te dirá qué estudia y/o su meta laboral. Debes responder con un plan de acción MUY BREVE de 3 puntos: 

            1) Una habilidad clave que deben aprender. 

            2) Un consejo práctico para su CV o LinkedIn. 

            3) Un consejo de networking o búsqueda de empleo. 

            Mantén la respuesta en un solo párrafo o lista corta. Usa emojis.`;


            const payload = {

                contents: [{ parts: [{ text: promptText }] }],

                systemInstruction: { parts: [{ text: systemPrompt }] }

            };


            const delays = [1000, 2000, 4000, 8000, 16000];

            

            // Exponential backoff retry logic

            for (let attempt = 0; attempt < 6; attempt++) {

                try {

                    const response = await fetch(url, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    

                    if (!response.ok) {

                        throw new Error(`Error HTTP: ${response.status}`);

                    }

                    

                    const data = await response.json();

                    return data.candidates?.[0]?.content?.parts?.[0]?.text || "Lo siento, no pude generar un plan en este momento.";

                    

                } catch (error) {

                    if (attempt === 5) {

                        console.error("Fallo definitivo de la API:", error);

                        return "Hubo un problema de conexión con nuestros servidores de IA. ¡Por favor intentá de nuevo más tarde!";

                    }

                    await new Promise(resolve => setTimeout(resolve, delays[attempt]));

                }

            }

        }


        async function generateCareerPlan() {

            // Format simple markdown (bold tags and new lines) to HTML safely

            const formattedHTML = llmResponse

                .replace(/\n/g, '<br>');


            // Display result

            outputText.innerHTML = formattedHTML;


            // Reset button state

            btnEl.disabled = false;

            btnEl.innerHTML = 'Generar Plan de Acción ✨';

        }

        function wrapLabel(label) {

            if (label.length <= 16) return label;

            let words = label.split(' ');

            let lines = [];

            let currentLine = '';

            words.forEach(word => {

                if ((currentLine + word).length > 16) {

                    if (currentLine) lines.push(currentLine.trim());

                    currentLine = word + ' ';

                } else {

                    currentLine += word + ' ';

                }

            });

            if (currentLine) lines.push(currentLine.trim());

            return lines;

        }
